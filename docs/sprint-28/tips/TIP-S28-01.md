```prisma
// ============ DAILY PRODUCTION PLAN ============

model DailyPlanStatus {
  ...
}

enum DailyPlanStatus {
  DRAFT
  APPROVED
  IN_PROGRESS
  COMPLETED
  CANCELLED
}

model DailyProductionPlan {
  id             String                    @id @default(cuid())
  planNumber     String                    @unique
  date           DateTime                  @db.Date
  workCenterId   String
  workCenter     WorkCenter                @relation(fields: [workCenterId], references: [id], onDelete: Restrict)
  status         DailyPlanStatus           @default(DRAFT)
  notes          String?
  createdByUserId String?
  createdByUser  User?                     @relation("DailyPlanCreator", fields: [createdByUserId], references: [id], onDelete: SetNull)
  approvedByUserId String?
  approvedByUser User?                     @relation("DailyPlanApprover", fields: [approvedByUserId], references: [id], onDelete: SetNull)
  approvedAt     DateTime?
  createdAt      DateTime                  @default(now())
  updatedAt      DateTime                  @updatedAt

  lines          DailyProductionPlanLine[]

  @@unique([date, workCenterId])
  @@index([date])
  @@index([workCenterId])
  @@index([status])
  @@map("daily_production_plans")
}

model DailyProductionPlanLine {
  id           String              @id @default(cuid())
  planId       String
  plan         DailyProductionPlan @relation(fields: [planId], references: [id], onDelete: Cascade)
  workOrderId  String
  workOrder    WorkOrder           @relation(fields: [workOrderId], references: [id], onDelete: Cascade)
  sequence     Int
  plannedQty   Int
  assignedToUserId String?
  assignedToUser User?             @relation("PlanLineAssignee", fields: [assignedToUserId], references: [id], onDelete: SetNull)
  estimatedStartTime String?       // "08:00"
  estimatedEndTime   String?       // "12:00"
  notes        String?
  createdAt    DateTime            @default(now())
  updatedAt    DateTime            @updatedAt

  @@unique([planId, sequence])
  @@index([planId])
  @@index([workOrderId])
  @@index([assignedToUserId])
  @@map("daily_production_plan_lines")
}

// ============ SHIFT REPORT (auto-aggregated) ============

model ShiftReport {
  id                  String     @id @default(cuid())
  date                DateTime   @db.Date
  shiftId             String
  shift               Shift      @relation(fields: [shiftId], references: [id], onDelete: Restrict)
  workCenterId        String?
  workCenter          WorkCenter? @relation(fields: [workCenterId], references: [id], onDelete: SetNull)

  // Aggregated metrics
  totalOutput         Int        @default(0)
  totalScrap          Int        @default(0)
  scrapRate           Float?     // computed
  totalLaborMinutes   Float      @default(0)
  totalDowntimeMinutes Float     @default(0)
  headcountPlanned    Int        @default(0)
  headcountActual     Int        @default(0)

  // Source refs (for drill-down)
  laborEntryIds       Json?      // array of LaborEntry id
  downtimeRecordIds   Json?      // array of DowntimeRecord id
  productionReceiptIds Json?     // array of ProductionReceipt id
  workOrderIds        Json?      // array of WorkOrder id

  generatedAt         DateTime   @default(now())
  generatedByUserId   String?
  generatedByUser     User?      @relation("ShiftReportGenerator", fields: [generatedByUserId], references: [id], onDelete: SetNull)
  notes               String?
  createdAt           DateTime   @default(now())
  updatedAt           DateTime   @updatedAt

  @@unique([date, shiftId, workCenterId])
  @@index([date])
  @@index([shiftId, date])
  @@index([workCenterId, date])
  @@map("shift_reports")
}

// ============ BRIDGE LAYER MVP ============

enum DataSourceType {
  EXCEL_UPLOAD
  GSHEET           // pending Sprint 29
  CSV
}

enum SyncTier {
  REALTIME
  POLL
  ONDEMAND
}

model DataSource {
  id             String         @id @default(cuid())
  code           String         @unique
  name           String
  description    String?
  type           DataSourceType @default(EXCEL_UPLOAD)
  ownerDept      String?        // "Sản xuất", "Mua hàng"...
  url            String?        // GSheet URL or upload reference
  fileId         String?        // Drive fileId or upload key
  syncTier       SyncTier       @default(ONDEMAND)
  status         String         @default("active") // active, paused, archived
  lastSyncAt     DateTime?
  lastSyncStatus String?        // success, fail, partial
  createdByUserId String?
  createdByUser  User?          @relation("DataSourceCreator", fields: [createdByUserId], references: [id], onDelete: SetNull)
  createdAt      DateTime       @default(now())
  updatedAt      DateTime       @updatedAt

  mappings       MappingRule[]
  syncJobs       SyncJob[]

  @@index([type])
  @@index([status])
  @@map("data_sources")
}

model MappingRule {
  id             String     @id @default(cuid())
  sourceId       String
  source         DataSource @relation(fields: [sourceId], references: [id], onDelete: Cascade)
  version        Int        @default(1)
  isActive       Boolean    @default(true)
  targetEntity   String     // "Part", "Supplier", "WorkOrder"...
  columnMappings Json       // { "Mã": "partNumber", "Tên": "name", ... }
  authorityFields Json?     // ["partNumber", "name", "supplier.name"]
  validations    Json?      // { partNumber: { required: true, regex: "..." } }
  createdByUserId String?
  createdByUser  User?      @relation("MappingRuleCreator", fields: [createdByUserId], references: [id], onDelete: SetNull)
  createdAt      DateTime   @default(now())
  updatedAt      DateTime   @updatedAt

  @@unique([sourceId, version])
  @@index([sourceId, isActive])
  @@map("mapping_rules")
}

model SyncJob {
  id             String     @id @default(cuid())
  sourceId       String
  source         DataSource @relation(fields: [sourceId], references: [id], onDelete: Cascade)
  mappingVersion Int
  status         String     @default("pending") // pending, running, done, fail
  rowsRead       Int        @default(0)
  rowsCreated    Int        @default(0)
  rowsUpdated    Int        @default(0)
  rowsConflict   Int        @default(0)
  rowsError      Int        @default(0)
  startedAt      DateTime?
  completedAt    DateTime?
  errorMessage   String?
  triggeredByUserId String?
  triggeredByUser User?     @relation("SyncJobTrigger", fields: [triggeredByUserId], references: [id], onDelete: SetNull)
  createdAt      DateTime   @default(now())
  updatedAt      DateTime   @updatedAt

  @@index([sourceId, status])
  @@index([status])
  @@index([createdAt])
  @@map("sync_jobs")
}
```

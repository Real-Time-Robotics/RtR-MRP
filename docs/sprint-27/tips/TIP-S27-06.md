Filter logic:
```ts
function filterGroups(groups: Group[], userRoles: RoleCode[]): Group[] {
  return groups
    .filter(g => g.roles === 'all' || userRoles.some(r => g.roles.includes(r)))
    .map(g => ({
      ...g,
      items: g.items.filter(i => !i.roles || userRoles.some(r => i.roles!.includes(r))),
    }));
}
```

### 3) Wire vào dashboard layout

**File:** `src/app/(dashboard)/layout.tsx` (hoặc nơi sidebar đang render).

```tsx
import { SidebarV2 } from '@/components/layout/sidebar-v2';
import { SidebarLegacy } from '@/components/layout/sidebar';  // tên hiện tại
import { FEATURE_FLAGS } from '@/lib/feature-flags';
import { auth } from '@/auth';
import { getUserRoles } from '@/lib/auth/rbac';

export default async function DashboardLayout({ children }: { children: React.ReactNode }) {
  const session = await auth();
  const userRoles = session?.user?.id ? await getUserRoles(session.user.id) : [];
  const Sidebar = FEATURE_FLAGS.SIDEBAR_V2 ? SidebarV2 : SidebarLegacy;
  return (
    <div className="flex">
      <Sidebar user={{ id: session.user.id, roles: userRoles }} />
      <main className="flex-1">{children}</main>
    </div>
  );
}
```

**Lưu ý:** `getUserRoles(userId)` chưa chắc tồn tại — TIP-05 có `hasRole()` nhưng có thể không có `getUserRoles`. Nếu chưa có → thêm helper vào `src/lib/auth/rbac.ts`:
```ts
export async function getUserRoles(userId: string): Promise<RoleCode[]>;
```
Implementation: query `prisma.userRole.findMany({ where: { userId }, include: { role: true }})` → map ra `code as RoleCode`.
